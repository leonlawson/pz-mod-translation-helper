module SWFancyLanterns {
    imports {
        Base
    }

    recipe Craft Grey Fancy Lantern
    {
        destroy Torch/HandTorch,
        keep [Recipe.GetItemTypes.Screwdriver],
        ElectronicsScrap=2,
        ElectricWire,
        Glue=1,
        Result:Grey_Fancy_Lantern,
        SkillRequired:Electricity=2,
        Time:200.0,
        Category:Electrical,
    }

    recipe Dismantle Grey Fancy Lantern
    {
        destroy Grey_Fancy_Lantern,
        keep [Recipe.GetItemTypes.Screwdriver],
        Result:ElectronicsScrap,
        Time:400.0,
        OnTest:Recipe.OnTest.DismantleElectronics,
        OnGiveXP:Recipe.OnGiveXP.DismantleElectronics,
        OnCreate:Recipe.OnCreate.DismantleTVRemote,
        Category:Electrical,
        AnimNode:Disassemble,
        Prop1:Source=2,
        Prop2:Source=1,
    }
}
