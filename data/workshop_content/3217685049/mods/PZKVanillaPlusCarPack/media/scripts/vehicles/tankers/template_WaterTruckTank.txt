  /***** Water Filter *******/
	template vehicle RS_WaterFilter
	{
		part RS_WaterFilter
		{
			specificItem = false,
			/*category = gastank,*/
			area = FuelStorage,
			itemType = Base.RS_WaterFilter,
			mechanicRequireKey = false,
			repairMechanic = true,
			lua
			{
				create = Vehicles.Create.Default,
			}
			
			table install
            {
                items
                {
                    item
                    {
                        type = Base.Wrench,
                        count = 1,
                        keep = true,
                        equip = primary,
                    }
                }
                time = 500,
                skills = Mechanics:3,
                recipes = Basic Mechanics,
                test = Vehicles.InstallTest.Default,
            }
            table uninstall
            {
                items
                {
                    item
                    {
                        type = Base.Wrench,
                        count = 1,
                        keep = true,
                        equip = primary,
                    }
                }
                time = 500,
                skills = Mechanics:3,
                recipes = Basic Mechanics,
                test = Vehicles.UninstallTest.Default,
            }
		}
	}
